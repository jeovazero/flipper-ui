import AutoComplete from '../core/AutoComplete'
import ListItem from '../core/ListItem'
import TextField from '../core/TextField'
import Typography from '../core/Typography'
import { Playground, PropsTable } from 'docz'

# AutoComplete

## Properties
<PropsTable of={ AutoComplete } />

## Simple
<Playground style={ { minHeight: '200px' } }>
    <AutoComplete
        data={ ['Rock', 'Pop', 'Classic'] }
        inputElement={
            <TextField
                fullWidth
                placeholder='Music genres'
            />
        }
        renderSuggestion={ (item, props) =>
            <ListItem key={ item } { ...props }>
                { item }
            </ListItem>
        }
    />
</Playground>

## With subheader
<Playground style={ { minHeight: '200px' } }>
    <AutoComplete
        data={ [{ label: 'Genres', subheader: true }, 'Rock', 'Pop', 'Classic'] }
        inputElement={
            <TextField
                fullWidth
                placeholder='Music genres'
            />
        }
        renderSuggestion={ (item, props) => item.subheader
            ? (
                <ListItem key={ item }>
                    <Typography variant='caption'>
                        { item.label }
                    </Typography>
                </ListItem>
            )
            : (
                <ListItem key={ item } { ...props }>
                    { item.label || item }
                </ListItem>
            )
        }
    />
</Playground>
