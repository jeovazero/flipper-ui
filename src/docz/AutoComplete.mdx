import AutoComplete from '../core/AutoComplete'
import ListItem from '../core/ListItem'
import TextField from '../core/TextField'
import Typography from '../core/Typography'
import { Playground, Props } from 'docz'

# AutoComplete

## Properties
<Props of={ AutoComplete } />

## Simple
<Playground style={ { minHeight: '200px' } }>
    <AutoComplete
        suggestions={ ['Rock', 'Pop', 'Classic'] }
        renderInput={ props =>
            <TextField
                fullWidth
                placeholder='Music genres'
                { ...props }
            />
        }
        renderSuggestion={ (item, props, selected) =>
            <ListItem
                key={ item }
                selected={ selected }
                { ...props }>
                { item }
            </ListItem>
        }
        onChange={ () => null }
    />
</Playground>

## With subheader
<Playground style={ { minHeight: '200px' } }>
    <AutoComplete
        suggestions={ [{ label: 'Genres', subheader: true }, 'Rock', 'Pop', 'Classic'] }
        defaultValue='Rock'
        renderInput={ props =>
            <TextField
                fullWidth
                placeholder='Music genres'
                { ...props }
            />
        }
        renderSuggestion={ (item, props, selected) => item.subheader
            ? (
                <ListItem key={ item }>
                    <Typography variant='caption'>
                        { item.label }
                    </Typography>
                </ListItem>
            )
            : (
                <ListItem
                    key={ item }
                    { ...props }
                    selected={ selected }>
                    { item.label || item }
                </ListItem>
            )
        }
        onChange={ () => null }
    />
</Playground>
